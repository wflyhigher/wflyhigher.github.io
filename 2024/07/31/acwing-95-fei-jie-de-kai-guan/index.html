AcWing 95. 费解的开关

发表于2024-07-31|更新于2024-07-31|acwing coding
|字数总计:620|阅读时长:2分钟|阅读量:

👉原题链接

题目描述: 题目表述较长，为加深理解，请移步至原题链接~

解题思路： 该题的解法有多种，个人觉得y总讲的这一方法十分巧妙，看完讲解，一时间感jio到醍醐灌顶，确实NB。大致思路如下：先规定不去变更第一行灯的状态，则要想把第一行中的0变为1的话，只能通过改变第一行中的0所对应的第二行的位置的灯的状态来实现，第三行、第四行以此类推，即第i + 1行该如何变化是由第i行的状态来决定的，但由于一共只有5行，因此最后一行无需再通过后面的行来做变更（因为它已经是最后一行了），最后只需通过判断最后一行是否全为1，进而确定当前状态是否合法。核心思想就是利用了递推。

具体代码：

#include <bits/stdc++.h>

using namespace std;

const int N = 10;

int T;
char g[N][N];
int dx[5]= {0, 1, 0, -1, 0}, dy[5] = {0, 0, -1, 0, 1};

void turn(int x, int y) // 把 '1' 变为 '0', '0' 变为 '1'
{
    for(int i = 0; i < 5; i++)
    {
        int a = x + dx[i], b = y + dy[i];
        if(a >= 0 && a < 5 && b >= 0 && b < 5)
            g[a][b] ^= 1;
    }
}

int fun()
{
    int ans = 0x3f3f3f3f;
    for(int k = 0; k < 1 << 5; k++) // 枚举第一行状态
    {
        int res = 0;
        char backup[N][N]; 
        memcpy(backup, g, sizeof g); // 拷贝当前状态，方便后续恢复现场
        for(int j = 0; j < 5; j++)
            if(k >> j & 1) // 此处是为1时，就先按一下，目的是保证操作32次，确保能找到最小值
            {
                res++;
                turn(0, j);
            }
            
        for(int i = 0; i < 4; i++) // 枚举0-3行，最后1行(即4行)不变，因为4行后没有其他行来改变4行的状态
        {
            for(int j = 0; j < 5; j++)
                if(g[i][j] == '0')
                {
                    turn(i + 1, j);
                    res++;
                }
        }
        
        bool flag = true;
        for(int j = 0; j < 5; j++) // 判断最后一行是否全为1，即状态是否合法
            if(g[4][j] == '0')
            {
                flag = false;
                break;
            }
        
        if(flag) ans = min(ans, res);
        
        memcpy(g, backup, sizeof backup);  // 恢复原状态，便于后续枚举下一个状态
    }
    
    if(ans > 6) return -1;
    return ans;
}

int main()
{
    cin >> T;
    while(T--)
    {
        for(int i = 0; i < 5; i++)
            cin >> g[i];
        cout << fun() << "\n";
    }
    
    return 0;
}
